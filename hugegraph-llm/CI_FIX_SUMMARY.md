- name: Run unit tests
  run: |
    source .venv/bin/activate
    export SKIP_EXTERNAL_SERVICES=true
    cd hugegraph-llm
    export PYTHONPATH="$(pwd)/src:$PYTHONPATH"
    
    # 跳过有问题的测试
    python -m pytest src/tests/ -v --tb=short \
